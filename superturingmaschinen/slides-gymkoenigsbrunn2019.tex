    \begin{center}\begin{tikzpicture}

    \item Es gibt auch nur \hil{$\boldsymbol{\aleph_0}$} viele \hil{ganze Zahlen}.

    \begin{center}\begin{tikzpicture}

    \item Ebenso gibt es nur \hil{$\boldsymbol{\aleph_0}$} viele
    \hil{rationale Zahlen}.

    \begin{center}\begin{tikzpicture}
